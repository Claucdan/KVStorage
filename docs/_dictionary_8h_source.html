80 Bucket<Key, Value> *_set(
81 std::shared_ptr<Key> key,
82 std::shared_ptr<Value> value,
83 Bucket<Key, Value> **pDictionary,
84 std::size_t dictionarySize) {
85 std::size_t index = getBucketIndex(*key, dictionarySize);
86
87 if (pDictionary[index] == nullptr) {
88 spdlog::debug("Creat new bucket with: key = {}, value = {} ", *key, *value);
89 pDictionary[index] = new Bucket{std::move(key), std::move(value)};
90 ++_dictionaryUsed;
91 return pDictionary[index];
92 }
93
94 Bucket<Key, Value> *pBucket = pDictionary[index];
95 while (pBucket != nullptr && *(pBucket->key) != *key) {
96 pBucket = pBucket->next;
97 }
98
99 if (pBucket != nullptr) {
100 pBucket->value = std::move(value);
101 return pBucket;
102 }
103
104 Bucket<Key, Value> *tmp = new Bucket{std::move(key), std::move(value)};
105 tmp->next = _dictionary[index];
106 pDictionary[index] = tmp;
107 ++_dictionaryUsed;
108 return tmp;
109 }

114 std::optional<Bucket<Key, Value> *> _get(Key key, Bucket<Key, Value> **pDictionary, std::size_t dictionarySize) {
115 std::size_t index = getBucketIndex(key, dictionarySize);
116
117 Bucket<Key, Value> *tmp = pDictionary[index];
118 while (tmp != nullptr && *(tmp->key) != key) {
119 tmp = tmp->next;
120 }
121
122 if (tmp == nullptr) {
123 spdlog::debug("Bucket was not found with: key = {}", key);
124 return std::nullopt;
125 }
126
127 spdlog::debug("Bucket was found with: key = {}, value = {}", key, *tmp->value);
128 return std::optional<Bucket<Key, Value> *>(tmp);

134 bool _remove(Key key, Bucket<Key, Value> **pDictionary, std::size_t dictionarySize) {
135 std::size_t index = getBucketIndex(key, dictionarySize);
136
137 Bucket<Key, Value> *current = pDictionary[index];
138 Bucket<Key, Value> *prev = nullptr;
139
140 while (current != nullptr && *current->key != key) {
141 prev = current;
142 current = current->next;
143 }
144
145 if (current == nullptr)
146 return false;
147
148 if (prev == nullptr) {
149 pDictionary[index] = current->next;
150 } else {
151 prev->next = current->next;
152 }
153
154 spdlog::debug("Delete bucket with: key = {}, value = {}", *current->key, *current->value);
155 delete current;
156 --_dictionaryUsed;
157 return true;
158 }

165 bool isNeedRehash() {
166 return _dictionarySize > BASIC_DICTIONARY_SIZE && _dictionaryUsed != 0 &&
167 (fillFactor() < DICTIONARY_LOW_LIMIT_REHASH || fillFactor() > DICTIONARY_HIGH_LIMIT_REHASH);
168 }

173 bool isRehashing() {
174 return _rehashIndex != -1;
175 }
176
180 void prepareRehash() {
181 if (fillFactor() < DICTIONARY_LOW_LIMIT_REHASH) {
182 _alternativeDictionarySize = _dictionarySize / 2;
183 } else {
184 _alternativeDictionarySize = _dictionarySize * 2;
185 }
186
187 _alternativeDictionary = new Bucket<Key, Value> *[_alternativeDictionarySize](nullptr);
188 _rehashIndex = 0;
189 }
190
194 void rehash() {
195 for (std::size_t i = 0;
196 i < DICTIONARY_REHASH_BUCKET_COUNT && _rehashIndex < _dictionarySize;
197 ++i, ++_rehashIndex) {
198
199 if (_dictionary[_rehashIndex] == nullptr) {
200 continue;
201 }
202
203 Bucket<Key, Value> *tmp = _dictionary[_rehashIndex];
204 Bucket<Key, Value> *next;
205 _dictionary[_rehashIndex] = nullptr;
206
207 while (tmp != nullptr) {
208 next = tmp->next;
209 delete tmp;
210 tmp = next;
211 }
212 }
213
214 if (_rehashIndex == _dictionarySize) {
215 _rehashIndex = -1;
216 delete[] _dictionary;
217 _dictionary = _alternativeDictionary;
218 _alternativeDictionary = nullptr;
219 }
220 }

226 void preCommandActions() {
227 if (isRehashing()) {
228 rehash();

232 if (isNeedRehash()) {
